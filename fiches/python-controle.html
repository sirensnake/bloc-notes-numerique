🔍 Conditions (if / elif / else)

age = 20
if age < 18:
    print("Mineur")
elif age < 65:
    print("Adulte")
else:
    print("Senior")

🔁 Boucles for

for i in range(5):
    print("Itération", i)

🔁 Boucles while

compteur = 0
while compteur < 3:
    print("Compteur:", compteur)
    compteur += 1

🛑 Contrôle de flux

for lettre in "Python":
    if lettre == "h":
        continue
    if lettre == "o":
        break
    print(lettre)
