🔒 Permissions de fichiers

ls -l                     # Affiche les permissions
chmod u+x fichier.sh     # Ajoute exécution au propriétaire
chmod 755 fichier.sh     # rwxr-xr-x
chown user:group fichier # Change le propriétaire

🛡️ Superutilisateur et sudo

sudo commande             # Exécute une commande en admin
sudo -i                  # Ouvre un shell root
visudo                   # Édite la configuration sudo

🔥 Pare-feu (ufw)

sudo ufw status           # État du pare-feu
sudo ufw enable           # Active ufw
sudo ufw allow 22         # Autorise SSH
sudo ufw deny 80          # Bloque HTTP
