Permissions de fichiers

ls -l                     # Affiche les permissions
chmod u+x fichier.sh     # Ajoute excution au propritaire
chmod 755 fichier.sh     # rwxr-xr-x
chown user:group fichier # Change le propritaire

Superutilisateur et sudo

sudo commande             # Excute une commande en admin
sudo -i                  # Ouvre un shell root
visudo                   # dite la configuration sudo

Pare-feu (ufw)

sudo ufw status           # tat du pare-feu
sudo ufw enable           # Active ufw
sudo ufw allow 22         # Autorise SSH
sudo ufw deny 80          # Bloque HTTP
