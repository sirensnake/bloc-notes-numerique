🔧 Définir une fonction

def dire_bonjour():
    print("Bonjour !")

📥 Passer des arguments

def saluer(nom):
    print(f"Salut {nom} !")

📤 Retourner une valeur

def addition(a, b):
    return a + b

🧰 Paramètres par défaut

def saluer(nom="ami"):
    print(f"Bonjour {nom} !")

📦 Arguments variables

def addition(*nombres):
    return sum(nombres)

🧠 Fonctions lambda

carre = lambda x: x * x

🔄 Fonctions imbriquées

def exterieur():
    def interieur():
        print("Fonction intérieure")
    interieur()
