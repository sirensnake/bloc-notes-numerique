🛠️ Définir une fonction

def saluer(nom):
    print(f"Bonjour {nom}!")

saluer("Guillaume")

📦 Retourner une valeur

def carre(x):
    return x * x

resultat = carre(5)
print(resultat)

🧭 Paramètres par défaut

def saluer(nom="ami"):
    print(f"Salut {nom}!")

⚙️ Lambda (fonction anonyme)

addition = lambda x, y: x + y
print(addition(3, 4))
