🏗️ Définir une classe

class Personne:
    def __init__(self, nom):
        self.nom = nom

    def saluer(self):
        print(f"Bonjour, je m'appelle {self.nom}.")

guillaume = Personne("Guillaume")
guillaume.saluer()

🔁 Héritage

class Employe(Personne):
    def travailler(self):
        print(f"{self.nom} travaille")

bob = Employe("Bob")
bob.travailler()

🔒 Encapsulation (convention)

class Secret:
    def __init__(self):
        self._cache = "mot de passe"  # Convention: _ signifie privé
