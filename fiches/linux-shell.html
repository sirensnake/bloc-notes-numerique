💬 Commandes de base

echo "Bonjour"         # Affiche du texte
read nom              # Lit une entrée utilisateur
echo $nom             # Affiche la valeur de la variable

📝 Création de scripts

#!/bin/bash

nom="Guillaume"
echo "Bonjour $nom"

Sauvegarde dans un fichier avec .sh puis exécution :

chmod +x monscript.sh
./monscript.sh

🔁 Structures de contrôle

# Boucle for
for i in 1 2 3; do
  echo $i
done

# Condition if
if [ $a -eq 5 ]; then
  echo "Cinq"
else
  echo "Pas cinq"
fi
