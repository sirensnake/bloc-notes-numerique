Gestion des utilisateurs

adduser nom_utilisateur          # Ajouter un utilisateur
passwd nom_utilisateur          # Modifier son mot de passe
usermod -aG sudo utilisateur    # Ajouter au groupe sudo
id nom_utilisateur              # Voir les groupes dun utilisateur

Groupes et appartenance

groupadd nom_du_groupe          # Crer un groupe
gpasswd -a utilisateur groupe  # Ajouter un utilisateur  un groupe
deluser utilisateur groupe     # Retirer dun groupe

Droits et permissions

chmod +x script.sh              # Ajouter droit d'excution
chmod 755 fichier              # Dfinir permissions numriques
chown user:group fichier       # Changer propritaire et groupe
