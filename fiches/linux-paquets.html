🔄 Mise à jour du système

sudo apt update               # Met à jour la liste des paquets
sudo apt upgrade             # Met à jour les paquets installés
sudo apt full-upgrade        # Mise à jour complète

📥 Installation de paquets

sudo apt install nom         # Installe un paquet
sudo apt install htop curl  # Installe plusieurs paquets

❌ Suppression

sudo apt remove nom          # Supprime un paquet (garde les confs)
sudo apt purge nom           # Supprime tout (inclus fichiers de conf)

🔍 Recherche et informations

apt search nom               # Recherche un paquet
apt show nom                # Détails du paquet installé

📦 Autres systèmes de paquets

# Snap (Ubuntu)
sudo snap install nom

# DNF (Fedora)
sudo dnf install nom

# Flatpak (optionnel)
flatpak install flathub org.gimp.GIMP
