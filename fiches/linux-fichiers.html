Commandes de base

pwd                    # Affiche le chemin courant
ls                     # Liste les fichiers du rpertoire
ls -l /etc             # Liste dtaille d'un dossier spcifique
cd /chemin             # Se dplacer dans un rpertoire
cd ~                   # Aller dans le rpertoire personnel

Cration et suppression

touch fichier.txt           # Crer un fichier vide
mkdir mon_dossier           # Crer un nouveau dossier
rm fichier.txt              # Supprimer un fichier
rm -r mon_dossier           # Supprimer un dossier et son contenu

Copie et dplacement

cp fichier.txt sauvegarde.txt        # Copier un fichier
cp -r dossier1 dossier2              # Copier un dossier
mv fichier.txt dossier/              # Dplacer un fichier

Liens symboliques

ln -s /chemin/fichier lien_symbolique    # Crer un lien symbolique

Affichage de contenu

cat fichier.txt         # Affiche le contenu complet
head -n 10 fichier.txt  # Affiche les 10 premires lignes
tail -n 10 fichier.txt  # Affiche les 10 dernires lignes
less fichier.txt        # Affichage pagin
